  <script>
    // IntersectionObserver to reveal elements smoothly
    (function(){
      const observer = new IntersectionObserver((entries)=>{
        entries.forEach(entry=>{
          if(entry.isIntersecting){
            entry.target.classList.add('show');
            // If you want one-time reveal, unobserve after shown:
            observer.unobserve(entry.target);
          }
        });
      }, { threshold: 0.16 });

      document.querySelectorAll('.reveal, .reveal-left, .reveal-right').forEach(el=>{
        // choose initial class based on presence of left/right classes
        if(el.classList.contains('reveal-left')) {
          el.classList.add('reveal-left');
        } else if(el.classList.contains('reveal-right')) {
          el.classList.add('reveal-right');
        } else {
          el.classList.add('reveal');
        }
        observer.observe(el);
      });
    })();

    // Gallery tilt / subtle reaction on scroll
    (function(){
      const imgs = document.querySelectorAll('.gimg');
      function updateTilt(){
        const viewportCenter = window.scrollY + (window.innerHeight / 2);
        imgs.forEach(el=>{
          const rect = el.getBoundingClientRect();
          const elCenter = rect.top + window.scrollY + rect.height/2;
          const diff = (elCenter - viewportCenter) / (window.innerHeight);
          const speed = parseFloat(el.dataset.tiltSpeed || 1);
          const rotate = Math.max(-12, Math.min(12, -diff * 18 * speed));
          const translateY = Math.max(-18, Math.min(18, -diff * 8 * speed));
          el.style.transform = `rotate(${rotate}deg) translateY(${translateY}px)`;
          el.style.boxShadow = `0 ${10 + Math.abs(diff*18)}px ${24 + Math.abs(diff*30)}px rgba(15,15,15,0.04)`;
        });
      }
      // initial
      updateTilt();
      let ticking = false;
      window.addEventListener('scroll', ()=> {
        if(!ticking){
          window.requestAnimationFrame(()=>{ updateTilt(); ticking=false; });
          ticking = true;
        }
      }, { passive:true });
      window.addEventListener('resize', updateTilt);
    })();

    // Smooth small page load fade-in
    window.addEventListener('load', ()=> {
      document.querySelectorAll('.reveal, .reveal-left, .reveal-right').forEach((el,i)=>{
        setTimeout(()=> el.classList.add('show'), 120 + i*70);
      });
    });

    // Optional: add hash scroll offset fix for header
    window.addEventListener('hashchange', ()=>{
      window.scrollBy(0, -20);
    });

  </script>
